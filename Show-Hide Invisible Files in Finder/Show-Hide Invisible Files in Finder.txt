tell application "Terminal"		set status to (do shell script "defaults read com.apple.finder AppleShowAllFiles")		ignoring case		if status is "NO" then			do shell script "defaults write com.apple.finder AppleShowAllFiles YES"		else			do shell script "defaults write com.apple.finder AppleShowAllFiles NO"		end if	end ignoring		try		tell application "Finder" to quit	end try	delay 3	tell application "Finder" to activate	end tell
